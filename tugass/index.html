Multiple Inheritance dengan Mixin

  • Multiple inheritance: digunakan agar satu objek mewarisi perilaku/properti dari dua atau lebih class sekaligus.
  • Dart: pakai mixin untuk menggabungkan fungsionalitas beberapa class tanpa pewarisan rumit.
  • Bahasa yang mendukung: C++, Python, Scala, Perl (multiple inheritance langsung), Java, C#, Kotlin (single inheritance).
  • Contoh: Bebek (berenang + terbang), Smartphone (kamera + telepon).

// Nama  : Putri Ananda Amelia

mixin CanFly {
  bool _isFlying = false;

  bool get isFlying => _isFlying;

  void fly() {
    if (!_isFlying) {
      _isFlying = true;
      print('Started flying!');
    } else {
      print('Already flying!');
    }
  }

  void land() {
    if (_isFlying) {
      _isFlying = false;
      print('Landed!');
    } else {
      print('Already on the ground!');
    }
  }
}

mixin CanSwim {
  bool _isSwimming = false;

  bool get isSwimming => _isSwimming;

  void swim() {
    if (!_isSwimming) {
      _isSwimming = true;
      print('Started swimming!');
    } else {
      print('Already swimming!');
    }
  }

  void stopSwim() {
    if (_isSwimming) {
      _isSwimming = false;
      print('Stop swimming!');
    } else {
      print('Already stopped swimming!');
    }
  }
}

class Bat with CanFly {}

class Frog with CanSwim {}

class FlyingFish with CanFly, CanSwim {}

class Dolphin with CanSwim {}

void main() {
  Bat bat = Bat();
  bat.fly();
  bat.land();

  print('Is Bat flying? : ${bat.isFlying}');

  print('----------------');

  Frog frog = Frog();
  frog.swim();
  frog.stopSwim();

  print('Is Frog swimming? : ${frog.isSwimming}');

  print('----------------');

  FlyingFish fish = FlyingFish();
  fish.swim();
  fish.fly();
  fish.land();
  fish.stopSwim();

  print('Is FlyingFish swimming? : ${fish.isSwimming}');

  print('----------------');

  Dolphin dolphin = Dolphin();
  dolphin.swim();

Sealed Class

  • Mendeklarasikan sealed class ApiResponse
  • Menentukan turunan: Succes dan Error
  • Berfungsi sebagai library (dasar class)

//libsealed.dart
sealed class ApiResponse {}

class Success extends ApiResponse {
  final String message;
  Success(this.message);
}

class Error extends ApiResponse {
  final String errorMessage;
  Error(this.errorMessage);
}

  • Menggunakan class dari libsealed.dart
  • Menampilkan hasil Succes dan Error
  • Di gunakan untuk menguji sealed class

// Kelas : SI2A

import 'libsealed.dart';

void handleResponse(ApiResponse response) {
  if (response is Success) {
    print("Success: ${response.message}");
  } else if (response is Error) {
    print("Error: ${response.errorMessage}");
  }
}

void main() {
  handleResponse(Success("Data berhasil"));
  handleResponse(Error("Terjadi kesalahan"));
}

Final Class

  • Mendeklarasikan final class Settings
  • Tidak bisa di-extend atau di-implement dari luar library Tapi bisa di-extend/implement di dalam library yang sama
  • Berguna untuk class yang ingin "ditutup" untuk modifikasi.

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

//libfinal.dart
final class Settings {
  final String theme;
  final int fontSize;

  Settings({required this.theme, required this.fontSize});

  void showSettings() {
    print("Theme: $theme, Font Size: $fontSize");
  }
}

base class CustomSettings extends Settings {
  CustomSettings(String theme, int fontSize)
      : super(theme: theme, fontSize: fontSize);
}
  • Membuat object dari Settings
  • Di gunakan untuk menguji final class

import 'libfinal.dart';

void main() {

  custom.showSettings();
}

Praktik 1 : Sealed Class dengan Pattern Matching

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
sealed class LoginStatus {
  const LoginStatus();
}

class LoggingIn extends LoginStatus {
  const LoggingIn();
}

class Success extends LoginStatus {
  final String username;
  const Success(this.username);
}

class Error extends LoginStatus {
  final String message;
  const Error(this.message);
}

void handleStatus(LoginStatus status) {
  switch (status) {
    case LoggingIn():
      print('Logging in...');
    case Success(:final username):
      print('Welcome, $username');
    case Error(:final message):
      print('Login failed: $message');
  }
}

void main() {
  final states = [
    LoggingIn(),
    Success('Putri'),
    Error('Wrong password'),
  ];

  for (var state in states) {
    handleStatus(state);
  }
}

Studi Kasus : Sistem State Management

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

sealed class DataState {}

class LoadingData extends DataState {}

class DataLoaded extends DataState {
  final List data;
  DataLoaded(this.data);
}

class DataError extends DataState {
  final String message;
  DataError(this.message);
}

void updateUI(DataState state) {
  switch (state) {
    case LoadingData():
      showLoading();
    case DataLoaded(:final data):
      showData(data);
    case DataError(:final message):
      showError(message);
  }
}

void showLoading() {
  print('⏳ Loading data...');
}

void showData(List data) {
  print('✅ Data loaded:');
  for (int i = 0; i < data.length; i++) {
    print('   ${i + 1}. ${data[i]}');
  }
}

void showError(String message) {
  print('❌ Error: $message');
}

void main() {
  final states = [
    LoadingData(),
    DataLoaded(['Data 1', 'Data 2']),
    DataError('Gagal memuat data'),
  ];

  for (var state in states) {
    updateUI(state);
  }
}

Kesimpulan

Sealed class cocok untuk membuat tipe data dengan hierarki terbatas agar pattern matching aman. Base class digunakan untuk membuat class dasar yang bisa diturunkan di framework atau aplikasi. Final class cocok untuk utility class yang tidak boleh diturunkan atau diimplementasi dari luar.

// math_utils.dart
final class MathUtils {
  static int tambah(int a, int b) {
    return a + b;
  }

  static int kurang(int a, int b) {
    return a - b;
  }

  static int kali(int a, int b) {
    return a * b;
  }

  static double bagi(int a, int b) {
    return a / b;
  }
}

  • Tidak perlu membuat objek
