Named Constructor

class Mobil {

 String? merk;
 int? tahun;

 Mobil(this.merk,this.tahun);

 Mobil.baru(){
  merk="Toyota";
  tahun=2023;
 }

 String toString(){
  return "Mobil $merk $tahun";
 }

}

void main(){

 var mobil1=Mobil("Honda",2020);
 print(mobil1);

 var mobil2=Mobil.baru();
 print(mobil2);

}
