Penjelasan Konstruktor dan Enkapsulasi

Dalam pemrograman berorientasi objek (OOP), terdapat beberapa konsep penting untuk mengatur struktur program agar lebih rapi dan mudah dipahami. Dua konsep yang sering digunakan adalah konstruktor dan enkapsulasi. Konstruktor merupakan method khusus dalam class yang digunakan untuk membuat atau menginisialisasi objek. Sedangkan enkapsulasi adalah teknik untuk melindungi data dalam class agar tidak dapat diakses langsung dari luar, biasanya menggunakan private variable, getter, dan setter.

Jenis-jenis Konstruktor Dart

Default Constructor, Named Constructor, Parameterized Constructor, Constant Constructor.

Named Constructor (Class Mobil)

class Mobil {
  String? merk;
  int? tahun;

  Mobil(this.merk, this.tahun);

  Mobil.baru() {
    merk = 'Toyota';
    tahun = 2023;
  }

  String toString() {
    return 'Mobil(merk: $merk, tahun: $tahun)';
  }
}

void main() {

  var mobilA = Mobil('Honda', 2020);
  print(mobilA);

  var mobilB = Mobil.baru();
  print(mobilB);

}

Constant Constructor (Class Warna)

class Warna {

  final int r,g,b;

  const Warna(this.r,this.g,this.b);

  String toString(){
    return 'Warna($r,$g,$b)';
  }

}

void main(){

  const merah = Warna(255,0,0);
  const merah2 = Warna(255,0,0);

  print(merah);
  print(merah2);

}

Initializer List (Class Titik)

class Titik {

  final double x,y;

  Titik(double x,double y)
      : x = x.abs(),
        y = y.abs();

}

void main(){

  var t1 = Titik(3,4);
  print(t1.x);

}

Getter

class Person {

  String _name;
  int age;

  Person(this._name,this.age);

  String get getName{
    return _name;
  }

}

void main(){

  var p = Person("Putri",19);
  print(p.getName);

}

Setter

class Person {

  String _name;
  int age;

  Person(this._name,this.age);

  set setName(String name){
    _name = name;
  }

  String getName(){
    return _name;
  }

}

void main(){

  var p = Person("Putri",19);

  p.setName = "Ananda";

  print(p.getName());

}

Tugas Bonus - Rekening Bank

class RekeningBank {

  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening,this._saldo);

  void deposit(double jumlah){
    if(jumlah>0){
      _saldo += jumlah;
    }
  }

  void tarik(double jumlah){
    if(jumlah <= _saldo){
      _saldo -= jumlah;
    }
  }

  void cekSaldo(){
    print(_nomorRekening);
    print(_saldo);
  }

}

void main(){

  var rekening = RekeningBank("4567890",400000);

  rekening.deposit(200000);
  rekening.tarik(100000);
  rekening.cekSaldo();

}
