class Mobil {
  String? merk;
  int? tahun;

  Mobil(this.merk, this.tahun);

  Mobil.baru() {
    merk = 'Toyota';
    tahun = 2023;
  }

  Mobil.dariJson(Map json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  String toString() {
    return 'Mobil(merk: $merk, tahun: $tahun)';
  }
}

void main() {

  var mobilA = Mobil('Honda', 2020);
  print(mobilA);

  var mobilB = Mobil.baru();
  print(mobilB);

  var mobilC = Mobil.dariJson({"merk":"Suzuki","tahun":2021});
  print(mobilC);

}

class Warna {
  final int r,g,b;

  const Warna(this.r,this.g,this.b);

  String toString(){
    return 'Warna(r:$r g:$g b:$b)';
  }
}

void main(){

  const merah=Warna(255,0,0);
  const merahLagi=Warna(255,0,0);

  print(merah);
  print(merahLagi);

}
